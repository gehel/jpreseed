16  package ch.ledcom.jpreseed;
17  
18  import java.io.IOException;
19  import java.net.URI;
20  import java.net.URISyntaxException;
21  import java.nio.file.Path;
22  import java.nio.file.Paths;
23  import java.util.Collections;
24  
25  public class JPreseed {
26  
27      public static final String JPRESEED_DIR = ".jpreseed";
28      private final CachedDownloader downloader;
29  
30      public JPreseed() {
31          String homeDir = System.getProperty("user.home");
32          Path cacheDirectory = Paths.get(homeDir, JPRESEED_DIR, "cache");
33          downloader = new CachedDownloader(new CacheNaming(cacheDirectory));
34      }
35  
36      public final void create(URI imageUrl) throws IOException {
37          new UsbCreator(
38                  downloader.download(imageUrl),
39                  Paths.get("boot.img"),
40                  Paths.get("syslinux.cfg"),
41                  Collections.<Path>emptySet()
42          ).create();
43      }
44  
45      public static void main(String[] args) throws URISyntaxException, IOException {
46          URI imageUrl = new URI("http://archive.ubuntu.com/ubuntu/dists/trusty-updates/main/installer-amd64/current/images/netboot/boot.img.gz");
47          new JPreseed().create(imageUrl);
48      }
49  
50  }
